zoom = alt.selection_interval(
    bind='scales',
    on="[mousedown[event.shiftKey], mouseup] > mousemove",
    translate="[mousedown[event.shiftKey], mouseup] > mousemove!",
    )


brush = alt.selection_interval(
    on="[mousedown[!event.shiftKey], mouseup] > mousemove",
    translate="[mousedown[!event.shiftKey], mouseup] > mousemove!",
    ) # had to consult stack overflow for this zooming and brush syntax. You can pan around when holding shift  


sc_points = alt.Chart(df).mark_point().encode(
    x='Attack',
    y='Speed',
    color=alt.when(brush).then(alt.value("steelblue")).otherwise(alt.value("grey"))
).properties(
    width=600,
    height=500,
    title= alt.Title(f'Attack and Speed Stats for Each Generation')
    ).add_params(
    brush, zoom, select3, 
    ).transform_filter(
        select3
    )

   

lbl = sc_points.mark_text(
    align='left',
    baseline='middle',
    dx=7,
).encode(
    text='Name'
)

ranked_text = alt.Chart(df).mark_text(align='right').encode(
y=alt.Y('row_number:O').axis(None)
).transform_filter(
brush, 
select3
).transform_window(
row_number='row_number()'
).transform_filter(
    alt.datum.row_number < 25
)

name = ranked_text.encode(text='Name:N').properties(
    title=alt.Title(text='Name', align='right')
)

chart = sc_points + lbl
text = alt.hconcat(name, primType, leg, hp, tbs)

fig = alt.hconcat(
